## 二、详细表结构

### 2.1 多租户表 (tenants)

```sql
-- 租户表
CREATE TABLE IF NOT EXISTS tenants (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name            VARCHAR(255) NOT NULL COMMENT '企业名称',
    short_name      VARCHAR(100) COMMENT '企业简称',
    logo            VARCHAR(500) COMMENT 'Logo URL',
    favicon         VARCHAR(500) COMMENT '网站图标',
    industry        VARCHAR(50) COMMENT '行业：manufacturing/electronics/equipment/other',
    scale           VARCHAR(20) COMMENT '规模：small/medium/large',
    contact         VARCHAR(100) COMMENT '联系人',
    phone           VARCHAR(20) COMMENT '联系电话',
    email           VARCHAR(100) COMMENT '企业邮箱',
    address         VARCHAR(500) COMMENT '企业地址',
    
    -- 套餐信息
    plan            VARCHAR(20) DEFAULT 'basic' COMMENT '套餐：basic/pro/enterprise',
    plan_expire_at  TIMESTAMP COMMENT '套餐过期时间',
    max_users       INTEGER DEFAULT 10 COMMENT '最大用户数',
    
    -- 状态
    status          VARCHAR(20) DEFAULT 'active' COMMENT '状态：active/disabled/trial',
    trial_expire_at TIMESTAMP COMMENT '试用期过期时间',
    
    -- 配置
    language        VARCHAR(10) DEFAULT 'zh-CN' COMMENT '默认语言',
    timezone        VARCHAR(50) DEFAULT 'Asia/Shanghai' COMMENT '时区',
    currency        VARCHAR(10) DEFAULT 'CNY' COMMENT '货币',
    theme_primary   VARCHAR(20) DEFAULT '#1E3A5F' COMMENT '主题主色',
    theme_accent    VARCHAR(20) DEFAULT '#0066FF' COMMENT '主题强调色',
    
    -- 平台开关
    enable_erp     BOOLEAN DEFAULT TRUE COMMENT '启用ERP对接',
    enable_audit    BOOLEAN DEFAULT TRUE COMMENT '启用审批流',
    
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    deleted_at     TIMESTAMP COMMENT '删除时间'
) COMMENT '租户表';

-- 索引
CREATE INDEX idx_tenants_status ON tenants(status);
CREATE INDEX idx_tenants_plan ON tenants(plan);
CREATE UNIQUE INDEX idx_tenants_name ON tenants(name) WHERE deleted_at IS NULL;
```

### 2.2 用户表 (users)

```sql
-- 用户表
CREATE TABLE IF NOT EXISTS users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
    
    -- 基本信息
    username       VARCHAR(50) NOT NULL COMMENT '用户名',
    password_hash  VARCHAR(255) NOT NULL COMMENT '密码哈希',
    real_name      VARCHAR(100) COMMENT '真实姓名',
    email          VARCHAR(100) COMMENT '邮箱',
    phone          VARCHAR(20) COMMENT '手机号',
    avatar         VARCHAR(500) COMMENT '头像URL',
    
    -- 角色与部门
    role           VARCHAR(20) NOT NULL DEFAULT 'buyer' COMMENT '角色：super_admin/tenant_admin/manager/buyer/finance',
    department     VARCHAR(100) COMMENT '部门',
    position       VARCHAR(100) COMMENT '职位',
    
    -- 状态
    status         VARCHAR(20) DEFAULT 'active' COMMENT '状态：active/disabled/locked',
    password_error_count INTEGER DEFAULT 0 COMMENT '密码错误次数',
    locked_until   TIMESTAMP COMMENT '锁定截止时间',
    last_login     TIMESTAMP COMMENT '最后登录时间',
    last_login_ip  VARCHAR(45) COMMENT '最后登录IP',
    
    -- 多因素认证
    mfa_enabled    BOOLEAN DEFAULT FALSE COMMENT '是否启用MFA',
    mfa_secret     VARCHAR(255) COMMENT 'MFA密钥',
    
    created_at     TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at     TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    deleted_at    TIMESTAMP COMMENT '删除时间',
    
    UNIQUE(tenant_id, username),
    UNIQUE(tenant_id, phone),
    UNIQUE(tenant_id, email)
) COMMENT '用户表';

-- 索引
CREATE INDEX idx_users_tenant ON users(tenant_id);
CREATE INDEX idx_users_role ON users(role);
CREATE INDEX idx_users_status ON users(status);
CREATE INDEX idx_users_phone ON users(phone);
```

### 2.3 角色权限表

```sql
-- 角色表
CREATE TABLE IF NOT EXISTS roles (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID REFERENCES tenants(id) ON DELETE CASCADE,
    name            VARCHAR(50) NOT NULL COMMENT '角色名称',
    code            VARCHAR(50) NOT NULL COMMENT '角色编码',
    description     VARCHAR(255) COMMENT '角色描述',
    is_system       BOOLEAN DEFAULT FALSE COMMENT '是否系统预置',
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(tenant_id, code)
) COMMENT '角色表';

-- 权限表
CREATE TABLE IF NOT EXISTS permissions (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name            VARCHAR(100) NOT NULL COMMENT '权限名称',
    code            VARCHAR(100) NOT NULL COMMENT '权限编码',
    resource        VARCHAR(50) NOT NULL COMMENT '资源：user/goods/purchase/approval/erp/config',
    action          VARCHAR(50) NOT NULL COMMENT '动作：create/read/update/delete/execute',
    description     VARCHAR(255) COMMENT '权限描述',
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(code)
) COMMENT '权限表';

-- 角色权限关联表
CREATE TABLE IF NOT EXISTS role_permissions (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    role_id         UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
    permission_id   UUID NOT NULL REFERENCES permissions(id) ON DELETE CASCADE,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(role_id, permission_id)
) COMMENT '角色权限关联表';

-- 用户角色关联表
CREATE TABLE IF NOT EXISTS user_roles (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    role_id         UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(user_id, role_id)
) COMMENT '用户角色关联表';
```

### 2.4 商品表

```sql
-- 商品分类表
CREATE TABLE IF NOT EXISTS goods_categories (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID REFERENCES tenants(id) ON DELETE CASCADE,
    parent_id       UUID REFERENCES goods_categories(id) ON DELETE CASCADE,
    name            VARCHAR(100) NOT NULL COMMENT '分类名称',
    code            VARCHAR(50) COMMENT '分类编码',
    sort            INTEGER DEFAULT 0 COMMENT '排序',
    status          VARCHAR(20) DEFAULT 'active' COMMENT '状态',
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(tenant_id, code)
) COMMENT '商品分类表';

-- 商品表
CREATE TABLE IF NOT EXISTS goods (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID REFERENCES tenants(id) ON DELETE CASCADE,
    category_id     UUID REFERENCES goods_categories(id) ON DELETE SET NULL,
    
    -- 基本信息
    name            VARCHAR(255) NOT NULL COMMENT '商品名称',
    code            VARCHAR(100) COMMENT '商品编码',
    barcode          VARCHAR(100) COMMENT '条形码',
    unit            VARCHAR(20) DEFAULT 'pcs' COMMENT '单位',
    
    -- 规格属性
    spec_template   JSONB COMMENT '规格模板JSON',
    spec_values     JSONB COMMENT '规格可选值JSON',
    
    -- 分类
    category_path   VARCHAR(500) COMMENT '分类路径： Electronics > Sensors',
    
    -- 图片
    images          JSONB COMMENT '图片URL数组',
    thumbnail       VARCHAR(500) COMMENT '缩略图',
    
    -- 描述
    description     TEXT COMMENT '商品描述',
    spec_description TEXT COMMENT '规格说明',
    
    -- 状态
    status          VARCHAR(20) DEFAULT 'active' COMMENT '状态：active/inactive',
    
    -- 扩展
    custom_fields   JSONB COMMENT '自定义字段',
    tags            JSONB COMMENT '标签数组',
    
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    deleted_at      TIMESTAMP COMMENT '删除时间',
    
    UNIQUE(tenant_id, code)
) COMMENT '商品表';

-- 商品规格表
CREATE TABLE IF NOT EXISTS goods_specs (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    goods_id        UUID NOT NULL REFERENCES goods(id) ON DELETE CASCADE,
    
    spec_name       VARCHAR(50) NOT NULL COMMENT '规格名称：Color/Size/Material',
    spec_value      VARCHAR(100) NOT NULL COMMENT '规格值：Red/M/L/Steel',
    
    -- 价格信息
    price           DECIMAL(12, 2) COMMENT '价格',
    cost_price      DECIMAL(12, 2) COMMENT '成本价',
    
    -- 库存信息
    stock_quantity  INTEGER DEFAULT 0 COMMENT '库存数量',
    min_stock       INTEGER DEFAULT 0 COMMENT '最小库存预警',
    
    -- SKU信息
    sku_code        VARCHAR(100) COMMENT 'SKU编码',
    barcode         VARCHAR(100) COMMENT '条形码',
    
    -- 状态
    status          VARCHAR(20) DEFAULT 'active' COMMENT '状态',
    
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(goods_id, spec_name, spec_value),
    UNIQUE(sku_code)
) COMMENT '商品规格表';

-- 索引
CREATE INDEX idx_goods_tenant ON goods(tenant_id);
CREATE INDEX idx_goods_category ON goods(category_id);
CREATE INDEX idx_goods_name ON goods(name);
CREATE INDEX idx_goods_code ON goods(code);
CREATE INDEX idx_goods_specs_goods ON goods_specs(goods_id);
```

```sql
-- 审批流程配置表
CREATE TABLE IF NOT EXISTS approval_workflows (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
    
    name            VARCHAR(100) NOT NULL COMMENT '流程名称',
    code            VARCHAR(50) NOT NULL COMMENT '流程编码',
    description     VARCHAR(255) COMMENT '流程描述',
    
    -- 流程配置
    workflow_type   VARCHAR(20) DEFAULT 'purchase' COMMENT '流程类型：purchase/contract/payment',
    trigger_amount DECIMAL(12, 2) COMMENT '触发金额阈值',
    
    -- 审批节点
    nodes            JSONB NOT NULL COMMENT '节点配置JSON',
    /* 
    nodes格式：
    [
      {
        "id": "node_1",
        "name": "部门经理审批",
        "type": "approver",
        "approver_type": "role",  // role/user/dynamic
        "approver_id": "manager",
        "timeout_hours": 24,
        "timeout_action": "auto_approve" // auto_approve/notify/escalate
      },
      {
        "id": "node_2", 
        "name": "ERP创建",
        "type": "sync",
        "target": "erp"
      }
    ]
    */
    
    is_active       BOOLEAN DEFAULT TRUE COMMENT '是否启用',
    is_system       BOOLEAN DEFAULT FALSE COMMENT '是否系统预置',
    
    created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(tenant_id, code)
) COMMENT '审批流程配置表';

-- 审批记录表
CREATE TABLE IF NOT EXISTS approval_records (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
```
