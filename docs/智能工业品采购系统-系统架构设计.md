### 4.2 核心表结构

#### 4.2.1 租户表 (tenants)

```sql
CREATE TABLE tenants (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name            VARCHAR(255) NOT NULL,           -- 企业名称
    short_name      VARCHAR(100),                     -- 企业简称
    logo            VARCHAR(500),                     -- Logo URL
    industry        VARCHAR(50),                      -- 行业
    scale           VARCHAR(20),                      -- 规模
    contact         VARCHAR(50),                     -- 联系人
    phone           VARCHAR(20),                      -- 电话
    address         VARCHAR(500),                    -- 地址
    plan            VARCHAR(20) DEFAULT 'basic',      -- 套餐: basic/pro/enterprise
    status          VARCHAR(20) DEFAULT 'active',     -- 状态: active/disabled
    language        VARCHAR(10) DEFAULT 'zh-CN',      -- 语言
    timezone        VARCHAR(50) DEFAULT 'Asia/Shanghai',
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_tenants_status ON tenants(status);
```

#### 4.2.2 用户表 (users)

```sql
CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID NOT NULL REFERENCES tenants(id),
    username        VARCHAR(50) NOT NULL,
    password_hash   VARCHAR(255) NOT NULL,
    real_name       VARCHAR(100),
    email           VARCHAR(100),
    phone           VARCHAR(20),
    role            VARCHAR(20) NOT NULL,            -- admin/manager/buyer
    status          VARCHAR(20) DEFAULT 'active',     -- active/disabled
    last_login      TIMESTAMP,
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW(),
    UNIQUE(tenant_id, username)
);

CREATE INDEX idx_users_tenant ON users(tenant_id);
CREATE INDEX idx_users_phone ON users(phone);
```

#### 4.2.3 采购申请表 (purchase_orders)

```sql
CREATE TABLE purchase_orders (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id           UUID NOT NULL REFERENCES tenants(id),
    order_no            VARCHAR(50) NOT NULL,         -- PR-YYYYMMDD-XXX
    user_id            UUID NOT NULL REFERENCES users(id),
    supplier_id        UUID,                        -- 选中的供应商
    goods_id           UUID,                        -- 商品ID
    goods_name         VARCHAR(255),
    specs              VARCHAR(500),                 -- 规格型号
    quantity           INTEGER NOT NULL,
    unit_price         DECIMAL(12, 2),              -- 单价
    total_amount       DECIMAL(12, 2),               -- 总金额
    delivery_time      DATE,                         -- 交货日期
    delivery_place     VARCHAR(255),                 -- 交货地点
    payment_method     VARCHAR(20),                  -- 付款方式
    project_name       VARCHAR(255),                 -- 项目名称
    usage_desc         VARCHAR(500),                 -- 用途说明
    department         VARCHAR(100),                 -- 需求部门
    status             VARCHAR(20) DEFAULT 'pending', -- draft/pending/approved/rejected/erp_synced
    erp_order_no      VARCHAR(50),                  -- ERP订单号
    approver_id        UUID,                         -- 审批人ID
    approved_at        TIMESTAMP,                    -- 审批时间
    reject_reason      VARCHAR(500),                 -- 驳回原因
    created_at         TIMESTAMP DEFAULT NOW(),
    updated_at         TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_orders_tenant ON purchase_orders(tenant_id);
CREATE INDEX idx_orders_status ON purchase_orders(status);
CREATE INDEX idx_orders_user ON purchase_orders(user_id);
CREATE UNIQUE INDEX idx_orders_no ON purchase_orders(tenant_id, order_no);
```

#### 4.2.4 价格缓存表 (price_cache)

```sql
CREATE TABLE price_cache (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    goods_sku           VARCHAR(100) NOT NULL,
    platform            VARCHAR(50) NOT NULL,         -- 平台: zkh/west/jd
    platform_goods_id   VARCHAR(100),
    price              DECIMAL(12, 2),
    stock_status       VARCHAR(20),                  -- in_stock/out_of_stock
    moq                INTEGER DEFAULT 1,             -- 最小起订量
    delivery_days      INTEGER,                      -- 交货天数
    cached_at           TIMESTAMP DEFAULT NOW(),
    expires_at         TIMESTAMP NOT NULL,
    UNIQUE(goods_sku, platform)
);

CREATE INDEX idx_price_cache_sku ON price_cache(goods_sku);
CREATE INDEX idx_price_cache_expires ON price_cache(expires_at);
```

### 5.3 安全措施

| 安全措施 | 实现方式 |
|----------|----------|
| 密码加密 | bcrypt + salt |

| CSRF 防护 | Token 验证 |

## 六、部署架构

### 6.1 Kubernetes 部署

```yaml
# Deployment 示例
apiVersion: apps/v1
kind: Deployment
metadata:
  name: user-service
  namespace: procurement
spec:
  replicas: 3
  selector:
    matchLabels:
      app: user-service
  template:
    spec:
      containers:
      - name: user-service
        image: registry.example.com/user-service:v1.0
        ports:
        - containerPort: 8001
        resources:
          requests:
            memory: "512Mi"
            cpu: "500m"
          limits:
            memory: "1Gi"
            cpu: "1000m"
        env:
        - name: DATABASE_URL
          valueFrom:
            secretKeyRef:
              name: db-credentials
              key: url
        - name: REDIS_URL
          valueFrom:
            configMapKeyRef:
              name: app-config
              key: redis_url
```
